Given a binary tree {1,2,3,4,5},

return the root of the binary tree [4,5,2,#,#,3,1].

   1
  /
 2   
/ \

4 5
／ ／ \
8 6 7 明白。

class Solution {
private:
stack> level;
public:
TreeNode upsideDownBinaryTree(TreeNode root) {
TreeNode first, second, new_root, node_temp;
pair temp = make_pair(root, NULL);
level.push(temp);
while (root->left) {
temp = make_pair(root->left, root->right);
level.push(temp);
root = root->left;
}
// 重建树
temp = level.top(); level.pop();
first = temp.first;
second = temp.second;
new_root = node_temp = first;
node_temp->left = second;
while (!level.empty()) {
temp = level.top(); level.pop();
first = temp.first;
second = temp.second;
node_temp->right = first;
node_temp = first;
node_temp->left = second;
}
}
return new_root;
};
